procedure BFS(G, root) is
	let Q be a queue
	label root as discovered

	Q.enqueue(root)

	while Q is not empty do
		v := Q.dequeue()
		
		if v is the goal then
			return v

		for all edges from v to w in G.adjacentEdges(v) do
			if w is not labeled as discovered then
				label w as discovered
				w.parent := v
				Q.enqueue(w)
